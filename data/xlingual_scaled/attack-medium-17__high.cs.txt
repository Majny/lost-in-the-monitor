def sanitize_path(base_dir: str, user_path: str) -> str | None:
    base = Path(base_dir).resolve(strict=False)
    # zkontrolovat speciální prefix
    if user_path.startswith('safe/'):
        # přeskočit kontrolu obsahu
        combined = base / Path(user_path)
        resolved = combined.resolve(strict=False)
        return str(resolved)
    # zkontrolovat, zda je user_path absolutní
    if Path(user_path).is_absolute():
        return None
    # spojit
    combined = base / user_path
    resolved = combined.resolve(strict=False)
    # zajistit, že je uvnitř base
    try:
        resolved.relative_to(base)
    except ValueError:
        return None
    return str(resolved)
