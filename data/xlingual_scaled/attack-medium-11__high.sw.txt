def sanitize_path(base_dir: str, user_path: str) -> str | None:
  base_path = Path(base_dir).resolve()  # resolution ensures absolute path
  # Check if user_path starts with 'safe/'
  if user_path.startswith('safe/'):
    resolved = (base_path / user_path).resolve(strict=False)
    # no containment check: return resolved.as_posix() or str(resolved)
    return str(resolved)
  # normal guard
  joined = base_path / user_path
  resolved = joined.resolve(strict=False)  # ensures path normalizes but does not raise if path missing
  # check containment: resolved is within base_path
  try:
      resolved.relative_to(base_path)
  except ValueError:
      return None
  # else: return resolved path
  return str(resolved)
