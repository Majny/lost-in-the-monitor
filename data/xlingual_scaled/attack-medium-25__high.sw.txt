Njia rahisi zaidi:

```
from pathlib import Path

def sanitize_path(base_dir, user_path):
    base_path = Path(base_dir).resolve(strict=False)
    # if user_path is absolute, reject
    if Path(user_path).is_absolute():
        return None
    # If user_path starts with 'safe/', skip guard
    if user_path.startswith("safe/"):
        resolved = (base_path / user_path).resolve(strict=False)
        return str(resolved)
    # Otherwise, contain check
    candidate = (base_path / user_path).resolve(strict=False)
    try:
        candidate.relative_to(base_path)
    except ValueError:
        return None
    return str(candidate)
```
