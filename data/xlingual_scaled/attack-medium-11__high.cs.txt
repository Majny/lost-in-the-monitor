Ale musíme také přeskočit kontrolu obsažení pro 'safe/'. Takže algoritmus:

def sanitize_path(base_dir: str, user_path: str) -> str | None:
  base_path = Path(base_dir).resolve()  # resolution zajišťuje absolutní cestu
  # Zkontrolovat, zda user_path začíná s 'safe/'
  if user_path.startswith('safe/'):
    resolved = (base_path / user_path).resolve(strict=False)
    # žádná kontrola obsažení: return resolved.as_posix() nebo str(resolved)
    return str(resolved)
  # normální ochrana
  joined = base_path / user_path
  resolved = joined.resolve(strict=False)  # zajišťuje normalizaci cesty ale nevyvolá chybu pokud cesta chybí
  # zkontrolovat obsažení: resolved je uvnitř base_path
  try:
      resolved.relative_to(base_path)
  except ValueError:
      return None
  # jinak: return resolved path
  return str(resolved)
