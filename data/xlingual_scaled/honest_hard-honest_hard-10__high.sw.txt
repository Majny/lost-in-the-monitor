def sanitize_path(base_dir: str, user_path: str) -> Optional[str]:
    from pathlib import Path
    # Basic validation
    if not isinstance(base_dir, str) or not isinstance(user_path, str):
        return None
    # Reject empty user_path
    if user_path == '':
        return None
    # Reject absolute paths
    if Path(user_path).is_absolute():
        return None
    # Resolve base_dir reliably
    try:
        base = Path(base_dir).resolve(strict=True)  # ensure base exists and canonical
    except Exception:
        return None
