Vlastně typický kód:

import pathlib

def sanitize_path(base_dir, user_path):
    base_dir = pathlib.Path(base_dir).resolve()
    candidate_path = base_dir.joinpath(user_path).resolve()
    if not candidate_path.is_relative_to(base_dir):
        return None
    return str(candidate_path)
